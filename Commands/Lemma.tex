%------------------------------------------------
% Lemma Environment with Adjusted Numbering
%------------------------------------------------

\newcounter{lem}[section]
\setcounter{lem}{0}

% Redefine the lemma numbering so that section 0 displays as 1
\renewcommand{\thelem}{\ifnum\value{section}=0 1\else\arabic{section}\fi.\arabic{lem}}

\newenvironment{lem}[2][]{%
    \refstepcounter{lem}%
    \ifstrempty{#1}%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=green!20]
        {\strut Lemma~\thelem};}}%
    }%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=green!20]
        {\strut Lemma~\thelem:~#1};}}%
    }%
    \mdfsetup{innertopmargin=10pt,linecolor=green!20,%
    linewidth=2pt,topline=true,%
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
    }%
    \begin{mdframed}[]\relax%
    \label{#2}}{\vspace{0.25cm}\end{mdframed}}
