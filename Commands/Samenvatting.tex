%------------------------------------------------
% Summary Environment with Adjusted Numbering
%------------------------------------------------

\newcounter{summ}[section]
\setcounter{summ}{0}

% Redefine the summary numbering so that section 0 displays as 1
\renewcommand{\thesumm}{\ifnum\value{section}=0 1\else\arabic{section}\fi.\arabic{summ}}

\newenvironment{summ}[2][]{%
    \refstepcounter{summ}%
    \ifstrempty{#1}%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=violet!20]
        {\strut Summary~\thesumm};}}%
    }%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=violet!20]
        {\strut Summary~\thesumm:~#1};}}%
    }%
    \mdfsetup{innertopmargin=10pt,linecolor=violet!20,%
    linewidth=2pt,topline=true,%
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
    }%
    \begin{mdframed}[]\relax%
    \label{#2}}{\vspace{0.25cm}\end{mdframed}}
