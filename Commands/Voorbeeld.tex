%------------------------------------------------
% Example Environment with Adjusted Numbering
%------------------------------------------------

\newcounter{ex}[section]
\setcounter{ex}{0}

% Redefine the example numbering so that section 0 displays as 1
\renewcommand{\theex}{\ifnum\value{section}=0 1\else\arabic{section}\fi.\arabic{ex}}

\newenvironment{ex}[2][]{%
    \refstepcounter{ex}%
    \ifstrempty{#1}%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=purple!20]
        {\strut Example~\theex};}}%
    }%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=purple!20]
        {\strut Example~\theex:~#1};}}%
    }%
    \mdfsetup{innertopmargin=10pt,linecolor=purple!20,%
    linewidth=2pt,topline=true,%
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
    }%
    \begin{mdframed}[]\relax%
    \label{#2}}{\vspace{0.25cm}\end{mdframed}}
