%------------------------------------------------
% Theorem Environment with Adjusted Numbering
%------------------------------------------------

\newcounter{theo}[section]
\setcounter{theo}{0}

% Redefine the theorem numbering so that section 0 displays as 1
\renewcommand{\thetheo}{\ifnum\value{section}=0 1\else\arabic{section}\fi.\arabic{theo}}

\newenvironment{theo}[2][]{%
    \refstepcounter{theo}%
    \ifstrempty{#1}%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=cyan!20]
        {\strut Theorem~\thetheo};}}%
    }%
    {\mdfsetup{%
        frametitle={%
        \tikz[baseline=(current bounding box.east),outer sep=0pt]
        \node[anchor=east,rectangle,fill=cyan!20]
        {\strut Theorem~\thetheo:~#1};}}%
    }%
    \mdfsetup{innertopmargin=10pt,linecolor=cyan!20,%
    linewidth=2pt,topline=true,%
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
    }%
    \begin{mdframed}[]\relax%
    \label{#2}}{\vspace{0.25cm}\end{mdframed}}
